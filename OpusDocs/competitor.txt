API.market Deep-Dive Report
Executive Summary
API.market is a developer-first API marketplace platform (not a single API service) that acts as a two-sided marketplace connecting API sellers (providers) with API buyers (developers/consumers). Think of it as "App Store for APIs" — a centralized platform to discover, test, integrate, and monetize APIs.

The Flow:
1. Sellers import APIs via OpenAPI/Swagger specs (YAML/JSON)
2. API.market proxies all requests — handles auth, billing, rate limiting, analytics
3. Buyers subscribe → get one unified API key → call any subscribed API
4. API.market takes a cut (revenue share model)

10. When to Use API.market
✅ Good Fit:

- Teams wanting unified billing/analytics

- Startups needing quick API integration without vendor negotiations

❌ Avoid If:
- Ultra-low latency required (<50ms critical)

- High-volume production where proxy cost + latency matters
- Need direct provider relationships/SLAs
- Custom auth flows (OAuth, mTLS) not supported by gateway

Artifact 1: Database Schema (db.js)
This sets up the SQLite state persistence for users, APIs, unified keys, subscriptions, and the usage logs required for the billing/revenue-share model.

javascript:
const Database = require('better-sqlite3');
const db = new Database('marketplace.db');

// Enable WAL mode for better concurrent read/write performance under gateway load
db.pragma('journal_mode = WAL');

db.exec(`
  CREATE TABLE IF NOT EXISTS users (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    email TEXT UNIQUE,
    role TEXT CHECK(role IN ('buyer', 'seller'))
  );

  CREATE TABLE IF NOT EXISTS apis (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    seller_id INTEGER,
    name TEXT,
    slug TEXT UNIQUE, -- e.g., 'openai', 'stable-diffusion'
    base_url TEXT,    -- Upstream API base URL
    upstream_auth_header TEXT, -- e.g., 'Authorization' or 'x-api-key'
    upstream_auth_token TEXT   -- The provider's actual secret key
  );

  CREATE TABLE IF NOT EXISTS api_keys (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER,
    key_value TEXT UNIQUE, -- The unified "mk_live_..." key
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
  );

  CREATE TABLE IF NOT EXISTS subscriptions (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER,
    api_id INTEGER,
    plan_tier TEXT, -- e.g., 'free', 'pro', 'enterprise'
    rate_limit_rpm INTEGER DEFAULT 60,
    UNIQUE(user_id, api_id)
  );

  CREATE TABLE IF NOT EXISTS usage_logs (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    api_key_id INTEGER,
    api_id INTEGER,
    status_code INTEGER,
    latency_ms INTEGER,
    timestamp DATETIME DEFAULT CURRENT_TIMESTAMP
  );
`);

module.exports = db;


Artifact 2: The Unified API Gateway (gateway.js)
This is the core proxy engine. It intercepts all buyer requests, validates the unified API key, checks subscriptions, enforces rate limits, proxies the request to the upstream provider, and logs the usage for analytics.

javascript:
const express = require('express');
const axios = require('axios');
const db = require('./db');

const router = express.Router();

// Middleware: Unified Auth, Subscription Check, and Rate Limiting
const gatewayMiddleware = (req, res, next) => {
  const authHeader = req.headers['authorization'];
  if (!authHeader?.startsWith('Bearer ')) {
    return res.status(401).json({ error: 'Missing Bearer token' });
  }

  const keyValue = authHeader.split(' ')[1];
  const apiSlug = req.params.apiSlug;

  // 1. Validate Unified API Key
  const keyRecord = db.prepare('SELECT * FROM api_keys WHERE key_value = ?').get(keyValue);
  if (!keyRecord) return res.status(401).json({ error: 'Invalid API Key' });

  // 2. Find the Target API
  const apiRecord = db.prepare('SELECT * FROM apis WHERE slug = ?').get(apiSlug);
  if (!apiRecord) return res.status(404).json({ error: 'API not found' });

  // 3. Check Subscription & Rate Limits
  const sub = db.prepare('SELECT * FROM subscriptions WHERE user_id = ? AND api_id = ?')
    .get(keyRecord.user_id, apiRecord.id);

  if (!sub) return res.status(403).json({ error: 'Subscription required for this API' });

  // Simple Rate Limiting Check (Requests per minute)
  const oneMinuteAgo = new Date(Date.now() - 60000).toISOString();
  const recentCalls = db.prepare(
    'SELECT COUNT(*) as count FROM usage_logs WHERE api_key_id = ? AND timestamp > ?'
  ).get(keyRecord.id, oneMinuteAgo);

  if (recentCalls.count >= sub.rate_limit_rpm) {
    return res.status(429).json({ error: 'Rate limit exceeded' });
  }

  // Attach context to request
  req.context = { keyRecord, apiRecord, sub };
  next();
};

// The Proxy Endpoint (Handles ALL HTTP methods)
router.all('/v1/proxy/:apiSlug/*', gatewayMiddleware, async (req, res) => {
  const { apiRecord, keyRecord } = req.context;

  // Extract the endpoint path (everything after /v1/proxy/{slug}/)
  const endpointPath = req.params[0];
  const targetUrl = `${apiRecord.base_url}/${endpointPath}${req._parsedUrl.search || ''}`;

  const startTime = Date.now();

  try {
    // Prepare headers for upstream provider
    const upstreamHeaders = { ...req.headers };
    delete upstreamHeaders['host'];
    delete upstreamHeaders['authorization']; // Strip buyer's unified key

    // Inject upstream provider's auth (managed secretly by your platform)
    if (apiRecord.upstream_auth_header) {
      upstreamHeaders[apiRecord.upstream_auth_header] = apiRecord.upstream_auth_token;
    }

    // Proxy the request
    const response = await axios({
      method: req.method,
      url: targetUrl,
      headers: upstreamHeaders,
      data: req.body,
      validateStatus: () => true, // Don't throw on 4xx/5xx from upstream
      timeout: 30000
    });

    const latency = Date.now() - startTime;

    // Log usage for analytics and billing (Revenue Share calculation)
    db.prepare(`
      INSERT INTO usage_logs (api_key_id, api_id, status_code, latency_ms)
      VALUES (?, ?, ?, ?)
    `).run(keyRecord.id, apiRecord.id, response.status, latency);

    // Return upstream response to the buyer
    res.status(response.status).send(response.data);

  } catch (error) {
    res.status(502).json({ error: 'Upstream API unreachable', details: error.message });
  }
});

module.exports = router;



Artifact 3: Seller OpenAPI Importer (seller.js)
Allows sellers to instantly onboard their APIs by simply providing a URL to their OpenAPI/Swagger spec.

const express = require('express');
const SwaggerParser = require('@apidevtools/swagger-parser');
const db = require('./db');

const router = express.Router();

router.post('/seller/import', async (req, res) => {
  const { specUrl, name, sellerId, upstreamAuthHeader, upstreamAuthToken } = req.body;

  try {
    // Parse and validate the OpenAPI spec automatically
    const api = await SwaggerParser.validate(specUrl);

    // Extract base URL from OpenAPI 3.0 'servers' array
    const baseUrl = api.servers?.[0]?.url;
    if (!baseUrl) throw new Error('OpenAPI spec must define a server URL');

    const slug = name.toLowerCase().replace(/[^a-z0-9]+/g, '-');

    // Save to database
    db.prepare(`
      INSERT INTO apis (seller_id, name, slug, base_url, upstream_auth_header, upstream_auth_token)
      VALUES (?, ?, ?, ?, ?, ?)
    `).run(sellerId, name, slug, baseUrl, upstreamAuthHeader || '', upstreamAuthToken || '');

    res.json({ success: true, slug, message: 'API imported and live!' });
  } catch (err) {
    res.status(400).json({ error: 'Failed to parse OpenAPI spec', details: err.message });
  }
});

module.exports = router;


Artifact 4: Analytics & Billing Queries
To implement the "Revenue Share" and "Analytics" features mentioned in the report, you can run these SQL queries against your usage_logs table:

javascript:
// Get real-time dashboard metrics for a specific API (Seller View)
const getApiAnalytics = (apiId) => {
  return db.prepare(`
    SELECT
      COUNT(*) as total_requests,
      ROUND(AVG(latency_ms), 2) as avg_latency_ms,
      SUM(CASE WHEN status_code >= 400 THEN 1 ELSE 0 END) as error_count,
      SUM(CASE WHEN status_code < 400 THEN 1 ELSE 0 END) as success_count
    FROM usage_logs
    WHERE api_id = ? AND timestamp > datetime('now', '-24 hours')
  `).get(apiId);
};

// Calculate Revenue Share (e.g., if you charge $0.001 per successful call and take 20%)
const calculateBilling = (apiId, pricePerCall, platformCutPercentage) => {
  const stats = db.prepare(`
    SELECT COUNT(*) as billable_calls
    FROM usage_logs
    WHERE api_id = ? AND status_code < 400 AND timestamp > datetime('now', '-30 days')
  `).get(apiId);

  const grossRevenue = stats.billable_calls * pricePerCall;
  const platformFee = grossRevenue * (platformCutPercentage / 100);
  const sellerPayout = grossRevenue - platformFee;

  return { grossRevenue, platformFee, sellerPayout, billableCalls: stats.billable_calls };
};
